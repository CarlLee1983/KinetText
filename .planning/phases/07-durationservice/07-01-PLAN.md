// DurationRequest - Bun 層批量元數據讀取請求
// filePaths: 要讀取的音頻檔案路徑陣列
// concurrency: 並發工作數（預設 4，允許 1-16）
type DurationRequest struct {
    FilePaths   []string `json:"file_paths"`
    Concurrency int      `json:"concurrency"` // optional, default 4
}

// DurationResponse - 元數據讀取結果
// success: 成功讀取的檔案數
// error: 第一個遇到的錯誤（若有）
// durations: { "/path/to/file.mp3": 120.5, ... } (秒)
type DurationResponse struct {
    Success   int              `json:"success"`
    Error     string           `json:"error,omitempty"`
    Durations map[string]float64 `json:"durations,omitempty"`
}

// FileMetadata - 單檔案元數據（內部使用）
type FileMetadata struct {
    Path     string  `json:"path"`
    Duration float64 `json:"duration"` // 秒
}

// ReaderConfig - 讀取器配置
type ReaderConfig struct {
    Concurrency     int           // 並發工作數（預設 4）
    TimeoutPerFile  time.Duration // 單檔案超時（預設 5 秒）
}

package main

import (
    "encoding/json"

    "os"

func main() {
    // 讀取標準輸入 (JSON)
    data, err := io.ReadAll(os.Stdin)
    if err != nil {
        respond(&DurationResponse{
            Success: 0,
            Error:   fmt.Sprintf("read stdin error: %v", err),
        })
        return
    }

    // 解析請求
    var req DurationRequest
    if err := json.Unmarshal(data, &req); err != nil {
        respond(&DurationResponse{
            Success: 0,
            Error:   fmt.Sprintf("JSON parse error: %v", err),
        })
        return
    }

    // 驗證請求
    if len(req.FilePaths) == 0 {
        respond(&DurationResponse{
            Success:   0,
            Durations: make(map[string]float64),
        })
        return
    }

    // 設定預設並發數
    concurrency := req.Concurrency
    if concurrency <= 0 || concurrency > 16 {
        concurrency = 4

    // 執行元數據讀取（實現在 reader.go）
    resp := ReadMetadataConcurrent(req.FilePaths, concurrency)
    respond(resp)

```go
package main

import (
    "context"
    "fmt"
    "os"
    "os/exec"
    "strconv"
    "strings"
    "sync"
    "time"

    "github.com/go-flac/flac-go/frame"
    "github.com/go-flac/flac-go/meta"
)

const (
    DefaultConcurrency = 4
    DefaultTimeout     = 5 * time.Second
    MaxConcurrency     = 16
)

// MetadataReader - 元數據讀取器介面
type MetadataReader interface {
    Read(ctx context.Context, filePath string) (float64, error)
}

// FLACReader - 使用 go-flac 讀取 FLAC 元數據
type FLACReader struct{}

func (r *FLACReader) Read(ctx context.Context, filePath string) (float64, error) {
    select {
    case <-ctx.Done():
        return 0, ctx.Err()
    default:
    }

    // 打開 FLAC 檔案
    file, err := os.Open(filePath)
    if err != nil {
        return 0, fmt.Errorf("open file: %w", err)
    }
    defer file.Close()

    // 解析 FLAC 元數據
    md, err := meta.Read(file)
    if err != nil {
        return 0, fmt.Errorf("read FLAC metadata: %w", err)
    }

    // 查詢 StreamInfo 區塊（包含採樣率和總樣本數）
    var duration float64
    for _, block := range md {
        if si, ok := block.(*meta.StreamInfo); ok {
            // Duration = TotalSamples / SampleRate
            if si.SampleRate == 0 {
                return 0, fmt.Errorf("invalid sample rate")
            }
            duration = float64(si.TotalSamples) / float64(si.SampleRate)
            return duration, nil
        }
    }

    return 0, fmt.Errorf("StreamInfo block not found")
}

// FFprobeReader - 使用 ffprobe 作為後備方案（MP3/AAC/OGG/FLAC 通用）
type FFprobeReader struct{}

func (r *FFprobeReader) Read(ctx context.Context, filePath string) (float64, error) {
    select {
    case <-ctx.Done():
        return 0, ctx.Err()
    default:
    }

    // 執行 ffprobe 取得時長
    // ffprobe -v error -show_entries format=duration -of default=noprint_wrappers=1:nokey=1:csv=p=0 <file>
    cmd := exec.CommandContext(
        ctx,
        "ffprobe",
        "-v", "error",
        "-show_entries", "format=duration",
        "-of", "default=noprint_wrappers=1:nokey=1:csv=p=0",
        filePath,
    )

    output, err := cmd.Output()
    if err != nil {
        return 0, fmt.Errorf("ffprobe error: %w", err)
    }

    // 解析輸出（秒數，浮點）
    durationStr := strings.TrimSpace(string(output))
    duration, err := strconv.ParseFloat(durationStr, 64)
    if err != nil {
        return 0, fmt.Errorf("parse duration: %w", err)
    }

    return duration, nil
}

// ReadMetadataConcurrent - 並發讀取多檔案元數據（公開函數，供 main.go 調用）
func ReadMetadataConcurrent(filePaths []string, concurrency int) *DurationResponse {
    if concurrency <= 0 || concurrency > MaxConcurrency {
        concurrency = DefaultConcurrency
    }

    ctx, cancel := context.WithTimeout(context.Background(), time.Duration(len(filePaths))*DefaultTimeout)
    defer cancel()

    return readWithWorkerPool(ctx, filePaths, concurrency)
}

// readWithWorkerPool - 使用 worker pool 並發讀取
func readWithWorkerPool(ctx context.Context, filePaths []string, concurrency int) *DurationResponse {
    durations := make(map[string]float64)
    var mu sync.Mutex
    var firstError string

    // 工作隊列
    jobs := make(chan string, len(filePaths))
    var wg sync.WaitGroup

    // 啟動 workers
    for i := 0; i < concurrency; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for filePath := range jobs {
                select {
                case <-ctx.Done():
                    return
                default:
                }

                duration, err := readMetadataWithFallback(ctx, filePath)
                mu.Lock()
                if err != nil {
                    if firstError == "" {
                        firstError = fmt.Sprintf("%s: %v", filePath, err)
                    }
                } else {
                    durations[filePath] = duration
                }
                mu.Unlock()
            }
        }()
    }

    // 排隊所有檔案
    for _, fp := range filePaths {
        jobs <- fp
    }
    close(jobs)

    // 等待所有 workers 完成
    wg.Wait()

    // 組裝回應
    resp := &DurationResponse{
        Success:   len(durations),
        Durations: durations,
    }
    if firstError != "" {
        resp.Error = firstError
    }
    return resp
}

// readMetadataWithFallback - 嘗試 FLAC 讀取，失敗則 fallback 至 ffprobe（D-01 決策）
func readMetadataWithFallback(ctx context.Context, filePath string) (float64, error) {
    // 檢查檔案是否存在
    if _, err := os.Stat(filePath); err != nil {
        return 0, fmt.Errorf("file not accessible: %w", err)
    }

    // 嘗試 FLAC 讀取
    flacReader := &FLACReader{}
    duration, err := flacReader.Read(ctx, filePath)
    if err == nil {
        return duration, nil
    }

    // FLAC 失敗，fallback 至 ffprobe（D-01）
    ffprobeReader := &FFprobeReader{}
    duration, err = ffprobeReader.Read(ctx, filePath)
    if err == nil {
        return duration, nil
    }

    // 兩者都失敗，返回錯誤（D-02 fail-fast）
    return 0, fmt.Errorf("FLAC read failed: %v; ffprobe fallback failed: %v", err, err)
}
```
