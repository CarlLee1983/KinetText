```typescript
import { spawn } from 'bun'
import { join } from 'node:path'
import { createLogger } from '../utils/logger'
import type { DurationGoConfig } from '../config/DurationGoConfig'

/**
 * Go 側 DurationRequest 型別（JSON）
 * 對應 kinetitext-go/src/duration-service/types.go
 */
interface GoReadMetadataRequest {
  file_paths: string[]
  concurrency?: number
}

/**
 * Go 側 DurationResponse 型別（JSON）
 */
interface GoReadMetadataResponse {
  success: number
  error?: string
  durations?: Record<string, number>
}

/**
 * DurationGoWrapper - Bun → Go subprocess JSON IPC
 *
 * 調用 kinetitext-go/bin/kinetitext-duration 並發讀取元數據
 * 支援 FLAC (go-flac) + MP3/AAC/OGG (ffprobe fallback)
 */
export class DurationGoWrapper {
  private static goBinary: string
  private static config: DurationGoConfig
  private static logger = createLogger('duration-go')

  static init(goBinaryPath: string, config: DurationGoConfig) {
    this.goBinary = goBinaryPath
    this.config = config
    this.logger.info({ goBinaryPath }, 'DurationGoWrapper initialized')
  }

  /**
   * 批量讀取元數據（Go 版本）
   * @param filePaths 檔案路徑陣列
   * @param concurrency 並發工作數（可選，預設 config.concurrency）
   * @returns Map<path, duration>，路徑 → 秒數
   * @throws 若 Go binary 不可用或 IPC 失敗
   */
  static async readMetadata(
    filePaths: ReadonlyArray<string>,
    concurrency: number = this.config.concurrency
  ): Promise<Map<string, number>> {
    if (!filePaths || filePaths.length === 0) {
      return new Map()
    }

    try {
      // 檢查 Go binary 存在
      if (!Bun.file(this.goBinary).exists()) {
        throw new Error(`Go binary not found: ${this.goBinary}`)
      }

      // 準備 JSON 請求
      const request: GoReadMetadataRequest = {
        file_paths: Array.from(filePaths),
        concurrency: Math.min(concurrency, 16),
      }

      const inputJson = JSON.stringify(request)

      // 執行 Go 程序
      const proc = spawn([this.goBinary], {
        stdin: 'pipe',
        stdout: 'pipe',
        stderr: 'pipe',
        timeout: this.config.timeout,
      })

      // 寫入 stdin
      await proc.stdin!.write(inputJson)
      proc.stdin!.end()

      // 等待完成並讀取 stdout
      const outputText = await proc.stdout!.text()
      const errorText = await proc.stderr!.text()
      const exitCode = await proc.exited

      if (exitCode !== 0) {
        throw new Error(
          `Go process exited with code ${exitCode}: ${errorText || outputText}`
        )
      }

      // 解析 Go 回應
      const response: GoReadMetadataResponse = JSON.parse(outputText)

      // 記錄成功/失敗
      this.logger.info(
        { success: response.success, total: filePaths.length, error: response.error },
        'Go metadata read completed'
      )

      // 轉換回 Map<string, number>
      const result = new Map<string, number>()
      if (response.durations) {
        for (const [path, duration] of Object.entries(response.durations)) {
          result.set(path, duration)
        }
      }

      // 若有錯誤但部分成功，記錄警告但不拋錯
      if (response.error && response.success > 0) {
        this.logger.warn({ error: response.error }, 'Partial metadata read success')
      }

      return result

    } catch (error) {
      const msg = error instanceof Error ? error.message : String(error)
      this.logger.error({ error: msg }, 'Go metadata read failed')

      // 拋出錯誤給調用者，讓 DurationService 決定是否 fallback
      throw error
    }
  }

  /**
   * 檢查 Go binary 是否可用
   */
  static isAvailable(): boolean {
    try {
      return Bun.file(this.goBinary).exists()
    } catch {
      return false
    }
  }
}

// 初始化（在 CrawlerEngine 或 main.ts 中調用）
export function initDurationGoBackend(config: DurationGoConfig) {
  const binaryPath = join(import.meta.dir, config.goBinaryPath)
  DurationGoWrapper.init(binaryPath, config)
}

export default DurationGoWrapper
```

```typescript
// 在 DurationService.ts 中做以下更改：

import { DurationGoWrapper } from '../services/DurationGoWrapper' // 新增 import
import type { DurationGoConfig } from '../config/DurationGoConfig'   // 新增 import

export interface DurationServiceDeps {
  metadataReader?: DurationMetadataReader
  goBackendConfig?: DurationGoConfig       // 新增：Go 後端配置
  enableGoBackend?: boolean                // 新增：啟用 Go 後端
}

export class DurationService {
  private readonly metadataReader: DurationMetadataReader
  private readonly goBackendEnabled: boolean
  private readonly logger = createLogger('duration')

  constructor(deps: DurationServiceDeps = {}) {
    this.metadataReader = deps.metadataReader ?? defaultMetadataReader
    this.goBackendEnabled = deps.enableGoBackend ?? false

    if (this.goBackendEnabled && deps.goBackendConfig) {
      // 初始化 Go 後端
      DurationGoWrapper.init(
        deps.goBackendConfig.goBinaryPath,
        deps.goBackendConfig
      )
    }
  }

  /**
   * 計算多檔案總時長
   * 新邏輯：優先嘗試 Go 後端（批量並發），失敗則 fallback 至 Bun Promise.all
   */
  async calculateTotalDuration(filePaths: ReadonlyArray<string>): Promise<number> {
    if (filePaths.length === 0) return 0

    // 優先嘗試 Go 後端（若啟用）
    if (this.goBackendEnabled && DurationGoWrapper.isAvailable()) {
      try {
        const startTime = performance.now()
        const durations = await DurationGoWrapper.readMetadata(filePaths)
        const elapsedMs = performance.now() - startTime

        // 驗證讀取結果
        if (durations.size > 0) {
          const total = Array.from(durations.values()).reduce((sum, d) => sum + d, 0)
          this.logger.info(
            {
              filesRead: durations.size,
              totalFiles: filePaths.length,
              totalDuration: total,
              elapsedMs: Math.round(elapsedMs)
            },
            'Go backend duration read completed'
          )

          // 若部分讀取失敗，補充用 Bun 讀取缺失檔案
          if (durations.size < filePaths.length) {
            const readPaths = new Set(durations.keys())
            const missingPaths = filePaths.filter(fp => !readPaths.has(fp))

            const missingDurations = await Promise.all(
              missingPaths.map(fp => this.getDuration(fp))
            )

            missingDurations.forEach((d, i) => {
              durations.set(missingPaths[i], d)
            })
          }

          return Array.from(durations.values()).reduce((sum, d) => sum + d, 0)
        }
      } catch (error) {
        this.logger.warn(
          { error: error instanceof Error ? error.message : String(error) },
          'Go backend failed, falling back to Bun'
        )
        // 繼續 fallback 至 Bun Promise.all
      }
    }

    // Fallback: 原 Bun 實現 (Promise.all)
    const durations = await Promise.all(
      filePaths.map(fp => this.getDuration(fp))
    )
    return durations.reduce((sum, d) => sum + d, 0)
  }

  /**
   * generateReport 也應該利用 Go 後端（可選優化）
   * 暫時保留原實現，Phase 7-02 可擴展
   */
  async generateReport(
    filePaths: ReadonlyArray<string>,
    targetSeconds: number = 39600,
    tolerancePercent: number = 10
  ): Promise<DurationReport> {
    const total = await this.calculateTotalDuration(filePaths)

    const files = await Promise.all(
      filePaths.map(async (fp) => ({
        path: fp,
        duration: await this.getDuration(fp),
      }))
    )

    // ... 保留原實現邏輯
    return {
      files,
      totalDuration: total,
      targetDuration: targetSeconds,
      withinTolerance: this.validateDuration(total, targetSeconds, tolerancePercent),
      tolerancePercent,
    }
  }
}
```
