2. **實現 converter.go** - FFmpeg 轉換核心:
   ```go
   package main

   import (
       "fmt"
       "os"
       "time"

       "github.com/u2takey/ffmpeg-go"
   )

   // ConvertAudio 執行音頻格式轉換
   func ConvertAudio(req AudioConvertRequest) (*AudioConvertResponse, error) {
       // 驗證輸入
       if req.InputFile == "" || req.OutputFile == "" {
           return nil, fmt.Errorf("input_file and output_file required")
       }

       // 檢查輸入文件存在
       if _, err := os.Stat(req.InputFile); os.IsNotExist(err) {
           return nil, fmt.Errorf("input file not found: %s", req.InputFile)
       }

       // 構建 FFmpeg 命令
       // ffmpeg-go API: ffmpeg.Input().Output().Compile()
       args := buildFFmpegArgs(req)

       // 執行轉換
       err := ffmpeg.Input(req.InputFile).
           Output(req.OutputFile, args...).
           Run()

       if err != nil {
           return &AudioConvertResponse{
               Success: false,
               Error:   fmt.Sprintf("FFmpeg error: %v", err),
           }, nil
       }

       // 驗證輸出文件
       if _, err := os.Stat(req.OutputFile); os.IsNotExist(err) {
           return &AudioConvertResponse{
               Success: false,
               Error:   "Output file not created",
           }, nil
       }

       // 獲取輸出文件大小（簡單驗證）
       info, _ := os.Stat(req.OutputFile)

       return &AudioConvertResponse{
           Success:    true,
           OutputFile: req.OutputFile,
           Duration:   0, // 可由 music-metadata 計算 (在 Bun 層)
       }, nil
   }

   // buildFFmpegArgs 構建 FFmpeg 命令行參數
   func buildFFmpegArgs(req AudioConvertRequest) []string {
       args := []string{}

       // 音頻編碼格式
       switch req.Format {
       case "mp3":
           args = append(args, "-c:a", "libmp3lame", "-q:a", "2") // 高質量
       case "aac":
           args = append(args, "-c:a", "aac", "-b:a", fmt.Sprintf("%dk", req.Bitrate))
       case "wav":
           args = append(args, "-c:a", "pcm_s16le")
       case "ogg":
           args = append(args, "-c:a", "libvorbis", "-b:a", fmt.Sprintf("%dk", req.Bitrate))
       case "flac":
           args = append(args, "-c:a", "flac")
       default:
           args = append(args, "-c:a", "libmp3lame") // 默認 MP3
       }

       // 位速率（針對 MP3、AAC、OGG）
       if req.Bitrate > 0 && (req.Format == "mp3" || req.Format == "aac" || req.Format == "ogg") {
           if req.Format != "mp3" { // MP3 已用 -q:a
               args = append(args, "-b:a", fmt.Sprintf("%dk", req.Bitrate))
           }
       }

       return args
   }
   ```

   import (
       "encoding/json"

   func main() {

       data, err := io.ReadAll(os.Stdin)

           return

       // 解析請求
       var req AudioConvertRequest
       if err := json.Unmarshal(data, &req); err != nil {
           respond(AudioConvertResponse{
               Success: false,
               Error:   fmt.Sprintf("JSON parse error: %v", err),
           })
           return
       }

       // 執行轉換
       resp, err := ConvertAudio(req)
       if err != nil {
           respond(AudioConvertResponse{
               Success: false,
               Error:   err.Error(),
           })
           return
       }

       // 返回響應
       respond(*resp)
   }

   func respond(resp AudioConvertResponse) {
       data, _ := json.Marshal(resp)
       fmt.Println(string(data))
   }

1. **Bun FFI 配置與包裝** (src/services/AudioConvertGoWrapper.ts):
   ```typescript
   import { FFIType, dlopen } from 'bun:ffi';
   import { join } from 'node:path';
   import { spawn } from 'bun';
   import type { AudioConvertRequest, AudioConvertResponse } from '@/types';

   /**
    * AudioConvertGoWrapper - Bun FFI 層包裝
    *
    * 通過 JSON 序列化和 spawn subprocess 調用 Go 二進制
    * 相比直接 FFI.cdef，使用 subprocess JSON 更穩定、跨平台
    */
   export class AudioConvertGoWrapper {
       private static goBinary: string;

       static init(goBinaryPath: string) {
           this.goBinary = goBinaryPath;
           // 驗證二進制存在
           if (!Bun.file(goBinary).exists()) {
               throw new Error(`Go binary not found: ${goBinaryPath}`);
           }
       }

       /**
        * 執行音頻轉換（Go 版本）
        * @param req AudioConvertRequest
        * @returns Promise<AudioConvertResponse>
        */
       static async convert(req: AudioConvertRequest): Promise<AudioConvertResponse> {
           try {
               // 準備 JSON 輸入
               const input = JSON.stringify({
                   input_file: req.inputFile,
                   output_file: req.outputFile,
                   format: req.format,
                   bitrate: req.bitrate || 192,
               });

               // 執行 Go 程序（子進程）
               const proc = spawn([this.goBinary], {
                   stdin: 'pipe',
                   stdout: 'pipe',
                   stderr: 'pipe',
               });

               // 寫入標準輸入
               await proc.stdin!.write(input);
               proc.stdin!.end();

               // 等待完成並讀取輸出
               const output = await proc.stdout?.text();
               const errorOutput = await proc.stderr?.text();
               const exitCode = await proc.exited;

               if (exitCode !== 0 || !output) {
                   return {
                       success: false,
                       error: `Go binary error (exit ${exitCode}): ${errorOutput}`,
                   };
               }

               // 解析 Go 程序返回的 JSON
               const resp = JSON.parse(output) as AudioConvertResponse;
               return resp;

           } catch (error) {
               return {
                   success: false,
                   error: `FFI wrapper error: ${error instanceof Error ? error.message : String(error)}`,
               };
           }
       }
   }

   // 初始化
   const goBinaryPath = join(import.meta.dir, '../../kinetitext-go/bin/kinetitext-audio');
   AudioConvertGoWrapper.init(goBinaryPath);

   export default AudioConvertGoWrapper;
   ```

**設計決策**:
- **IPC 協議**: 使用 subprocess JSON 而非 FFI.cdef
  - 原因: 穩定性更好，跨平台兼容 (macOS/Linux/Windows)
  - FFI.cdef 需要 C ABI 綁定，Go 輸出 cgo 二進制較複雜
  - JSON subprocess 方案成熟、易偵錯
- **進程管理**: 無連接池，每次調用啟動新進程
  - 優點: 無狀態，易於水平擴展（Bun 層可用 p-limit 控制並行度）
  - 缺點: 進程啟動開銷 ~1ms (可接受，相比原 100ms subprocess overhead 已大幅減少)
