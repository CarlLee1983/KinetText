2. **升級 AudioConvertService**:
   ```typescript
   // src/services/AudioConvertService.ts

   export class AudioConvertService {
       private config: AudioConvertConfig;
       private retryService: RetryService;
       private audioErrorClassifier: AudioErrorClassifier;
       private goWrapper?: AudioConvertGoWrapper;

       constructor(
           config: AudioConvertConfig,
           retryService?: RetryService,
           errorClassifier?: AudioErrorClassifier,
       ) {
           this.config = config;
           this.retryService = retryService || new RetryService();
           this.audioErrorClassifier = errorClassifier || new AudioErrorClassifier();

           // 初始化 Go 後端（如果啟用）
           if (config.useGoBackend && config.goBinaryPath) {
               try {
                   AudioConvertGoWrapper.init(config.goBinaryPath);
                   this.goWrapper = AudioConvertGoWrapper;
               } catch (error) {
                   console.warn('Failed to init Go backend, falling back to Bun:', error);
               }
           }
       }

       /**
        * 主轉換方法 - 支持後端選擇
        */
       async convert(req: AudioConvertRequest): Promise<AudioConvertResult> {
           const convertFn = this.config.useGoBackend && this.goWrapper
               ? () => this.convertWithGo(req)
               : () => this.convertWithBun(req);

           // 包裹在 RetryService 中
           return this.retryService.execute(convertFn, {
               maxRetries: this.config.maxRetries,
               initialDelayMs: this.config.initialRetryDelayMs,
               backoffMultiplier: this.config.retryBackoffMultiplier,
               classifier: this.audioErrorClassifier,
           });
       }

       /**
        * Bun FFmpeg-Simplified 實現（原有邏輯）
        */
       private async convertWithBun(req: AudioConvertRequest): Promise<AudioConvertResult> {
           // ... 現有的 FFmpeg-Simplified 邏輯保持不變
           return this.executeFFmpeg(req);
       }

       /**
        * Go kinetitext-audio 實現（新增）
        */
       private async convertWithGo(req: AudioConvertRequest): Promise<AudioConvertResult> {
           if (!this.goWrapper) {
               throw new Error('Go backend not initialized');
           }

           // 調用 Go wrapper
           const goResp = await this.goWrapper.convert(req);

           if (!goResp.success) {
               // 返回失敗結果，let RetryService 決定是否重試
               throw new Error(goResp.error || 'Go conversion failed');
           }

           // 驗證輸出文件存在
           if (!await this.verifyOutputFile(goResp.outputFile!)) {
               throw new Error('Output file not created or invalid');
           }

           return {
               success: true,
               outputFile: goResp.outputFile!,
               format: req.format,
               duration: goResp.duration,
               timestamp: Date.now(),
           };
       }

       /**
        * 驗證輸出文件
        */
       private async verifyOutputFile(filePath: string): Promise<boolean> {
           try {
               const file = await Bun.file(filePath).exists();
               if (!file) return false;

               // 基本驗證：檔案大小 > 0
               const stat = await Bun.file(filePath).stat?.();
               return stat ? stat.size > 0 : false;
           } catch {
               return false;
           }
       }
   }
   ```

1. **性能基準測試** (src/tests/integration/PerformanceBench.ts):
   ```typescript
   import { test, describe } from 'bun:test';
   import AudioConvertService from '@/services/AudioConvertService';
   import { generateAudioFiles } from './e2e-fixtures';
   import { join } from 'node:path';
   import { mkdtemp } from 'node:fs/promises';

   interface BenchmarkResult {
       name: string;
       backend: 'bun' | 'go';
       format: string;
       inputSize: number;
       duration: number; // ms
       success: boolean;
       error?: string;
   }

   export class AudioConvertBenchmark {
       private results: BenchmarkResult[] = [];

       /**
        * 運行性能基準測試
        * 對比 Bun 和 Go 後端在不同格式上的轉換速度
        */
       async benchmarkGoVsBun() {
           const testDir = await mkdtemp('/tmp/audio-bench-');
           const formats = ['wav', 'aac', 'ogg', 'flac'];

           for (const format of formats) {
               console.log(`\n📊 Benchmarking ${format}...`);

               // 生成測試音頻（3 秒，用於快速測試）
               const inputFile = await generateAudioFiles(testDir, [format]);

               // Bun 版本基準
               await this.benchmarkBackend('bun', inputFile[0], 'mp3');

               // Go 版本基準
               await this.benchmarkBackend('go', inputFile[0], 'mp3');
           }

           // 生成報告
           this.generateReport();
       }

       /**
        * 基準測試單個後端
        */
       private async benchmarkBackend(
           backend: 'bun' | 'go',
           inputFile: string,
           outputFormat: string
       ) {
           const outputFile = inputFile.replace(/\.[^.]+$/, `.${backend}.${outputFormat}`);
           const config = {
               useGoBackend: backend === 'go',
               goBinaryPath: '../../../kinetitext-go/bin/kinetitext-audio',
               maxRetries: 1,
           };

           const service = new AudioConvertService(config);

           // 測試 3 次取平均
           const times: number[] = [];
           for (let i = 0; i < 3; i++) {
               const start = performance.now();
               try {
                   await service.convert({
                       inputFile,
                       outputFile: `${outputFile}.${i}`,
                       format: outputFormat,
                       bitrate: 192,
                   });
                   times.push(performance.now() - start);
               } catch (err) {
                   this.results.push({
                       name: `${backend}-${outputFormat}-attempt${i}`,
                       backend,
                       format: outputFormat,
                       inputSize: Bun.file(inputFile).size,
                       duration: 0,
                       success: false,
                       error: String(err),
                   });
               }
           }

           if (times.length > 0) {
               const avgTime = times.reduce((a, b) => a + b, 0) / times.length;
               this.results.push({
                   name: `${backend}-${outputFormat}`,
                   backend,
                   format: outputFormat,
                   inputSize: Bun.file(inputFile).size,
                   duration: Math.round(avgTime),
                   success: true,
               });

               console.log(`  ✅ ${backend}: ${Math.round(avgTime)}ms (avg of ${times.length} runs)`);
           }
       }

       /**
        * 生成性能報告
        */
       private generateReport() {
           // 分組結果
           const grouped = new Map<string, BenchmarkResult[]>();
           for (const r of this.results) {
               const key = r.format;
               if (!grouped.has(key)) grouped.set(key, []);
               grouped.get(key)!.push(r);
           }

           // 計算改善百分比
           let markdown = `# 性能基準測試報告\n\n`;
           markdown += `**測試日期**: ${new Date().toISOString()}\n\n`;
           markdown += `## 概覽\n\n`;

           let totalGoImprovement = 0;
           let formatCount = 0;

           for (const [format, results] of grouped) {
               const bunResult = results.find(r => r.backend === 'bun');
               const goResult = results.find(r => r.backend === 'go');

               if (bunResult && goResult) {
                   const improvement = ((bunResult.duration - goResult.duration) / bunResult.duration * 100).toFixed(1);
                   totalGoImprovement += parseFloat(improvement);
                   formatCount++;

                   markdown += `| ${format.toUpperCase()} | ${bunResult.duration}ms | ${goResult.duration}ms | **${improvement}%** |\n`;
               }
           }

           markdown += `\n**平均提升**: ${(totalGoImprovement / formatCount).toFixed(1)}%\n`;
           markdown += `**性能目標**: 30-50% (${totalGoImprovement / formatCount > 30 ? '✅ 達成' : '❌ 未達成'})\n`;

           // 詳細結果表格
           markdown += `\n## 詳細結果\n\n`;
           markdown += `| 後端 | 格式 | 時間 | 結果 |\n`;
           markdown += `|------|------|------|------|\n`;

           for (const r of this.results) {
               markdown += `| ${r.backend} | ${r.format} | ${r.duration}ms | ${r.success ? '✅' : '❌'} |\n`;
           }

           // 寫入報告
           const reportPath = '.planning/phases/06-audio-convert-go/PERF_REPORT.md';
           Bun.write(reportPath, markdown);
           console.log(`\n📄 Report saved to: ${reportPath}`);
       }
   }
   ```
