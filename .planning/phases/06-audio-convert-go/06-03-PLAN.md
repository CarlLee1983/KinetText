   import { test, describe, beforeAll, afterAll, expect } from 'bun:test';
   import AudioConvertService from '@/services/AudioConvertService';
   import { generateAudioFile, verifyAudioFile } from './fixtures';
   import { mkdtemp } from 'node:fs/promises';
   import { join } from 'node:path';
   import pLimit from 'p-limit';

   describe('E2E: AudioConvert with Go Backend', () => {
       let testDir: string;
       const formats = ['wav', 'aac', 'ogg', 'flac'];
       let service: AudioConvertService;

       beforeAll(async () => {
           testDir = await mkdtemp('/tmp/audio-e2e-go-');

           // 初始化服務（啟用 Go 後端）
           service = new AudioConvertService({
               useGoBackend: true,
               goBinaryPath: '../../../kinetitext-go/bin/kinetitext-audio',
               maxRetries: 2,
               initialRetryDelayMs: 100,
           });
       });

       for (const format of formats) {
           test(`converts ${format} to MP3 via Go backend`, async () => {
               // 生成測試音頻（3 秒）
               const inputFile = await generateAudioFile(testDir, format, 3);
               const outputFile = join(testDir, `output-${format}.mp3`);

               // 執行轉換
               const result = await service.convert({
                   inputFile,
                   outputFile,
                   format: 'mp3',
                   bitrate: 192,
               });

               // 驗證結果
               expect(result.success).toBe(true);
               expect(result.outputFile).toBe(outputFile);

               // 驗證輸出文件
               const verified = await verifyAudioFile(outputFile);
               expect(verified).toBeDefined();
               expect(verified.duration).toBeDefined();
               expect(verified.format).toBe('mp3');
           });

           test(`converts ${format} to MP3 with correct duration`, async () => {
               const inputFile = await generateAudioFile(testDir, format, 5);
               const outputFile = join(testDir, `duration-${format}.mp3`);

               const result = await service.convert({
                   inputFile,
                   outputFile,
                   format: 'mp3',
               });

               expect(result.success).toBe(true);

               // 驗證時長（允許 ±1 秒誤差）
               const verified = await verifyAudioFile(outputFile);
               expect(Math.abs(verified.duration! - 5.0)).toBeLessThan(1.0);
           });
       }

       test('handles concurrent conversions with p-limit', async () => {
           const limit = pLimit(4);
           const tasks = [];

           for (let i = 0; i < 10; i++) {
               tasks.push(limit(async () => {
                   const format = formats[i % formats.length];
                   const input = await generateAudioFile(testDir, format, 2);
                   const output = join(testDir, `concurrent-${i}.mp3`);

                   const result = await service.convert({
                       inputFile: input,
                       outputFile: output,
                       format: 'mp3',
                   });

                   expect(result.success).toBe(true);
                   return result;
               }));
           }

           const results = await Promise.all(tasks);
           expect(results).toHaveLength(10);
           expect(results.every(r => r.success)).toBe(true);
       });

       test('reports meaningful errors for invalid input', async () => {
           const result = await service.convert({
               inputFile: '/nonexistent/file.wav',
               outputFile: '/tmp/out.mp3',
               format: 'mp3',
           });

           expect(result.success).toBe(false);
           expect(result.error).toBeDefined();
           expect(result.error).toContain('not found');
       });

       test('compares quality: Bun vs Go output', async () => {
           // 使用相同的輸入文件，分別用 Bun 和 Go 後端轉換
           const inputFile = await generateAudioFile(testDir, 'wav', 3);
           const bunOutput = join(testDir, 'bun-output.mp3');
           const goOutput = join(testDir, 'go-output.mp3');

           // Bun 版本
           const bunService = new AudioConvertService({ useGoBackend: false });
           const bunResult = await bunService.convert({
               inputFile,
               outputFile: bunOutput,
               format: 'mp3',
               bitrate: 192,
           });

           // Go 版本
           const goResult = await service.convert({
               inputFile,
               outputFile: goOutput,
               format: 'mp3',
               bitrate: 192,
           });

           // 兩者都應該成功
           expect(bunResult.success).toBe(true);
           expect(goResult.success).toBe(true);

           // 驗證兩個輸出都是有效的 MP3
           const bunVerified = await verifyAudioFile(bunOutput);
           const goVerified = await verifyAudioFile(goOutput);

           expect(bunVerified.format).toBe('mp3');
           expect(goVerified.format).toBe('mp3');

           // 時長應一致（±1 秒）
           const durationDiff = Math.abs(bunVerified.duration! - goVerified.duration!);
           expect(durationDiff).toBeLessThan(1.0);

           // 文件大小應接近（均為 192kbps）
           const bunSize = Bun.file(bunOutput).size;
           const goSize = Bun.file(goOutput).size;
           const sizeDiff = Math.abs(bunSize - goSize) / Math.min(bunSize, goSize);
           expect(sizeDiff).toBeLessThan(0.1); // 允許 10% 偏差
       });

       afterAll(async () => {
           // 清理測試目錄
           // rm -rf testDir
       });
   });

   import { join } from 'node:path';
   import { parseFile } from 'music-metadata';

   /**
    * 生成測試音頻文件（支持多格式）
    */
   export async function generateAudioFile(
       dir: string,
       format: string,
       durationSeconds: number
   ): Promise<string> {
       const outputFile = join(dir, `test-${Date.now()}.${format}`);

       // 使用 FFmpeg 生成靜音音頻
       // 相同的方式用於所有格式
       const cmd = `ffmpeg -f lavfi -i anullsrc=r=44100:cl=mono -t ${durationSeconds} -q:a 9 "${outputFile}" -y`;

       await Bun.shell(cmd).quiet().run();

       if (!Bun.file(outputFile).exists()) {
           throw new Error(`Failed to generate ${format} file`);
       }

       return outputFile;
   }

   /**
    * 驗證音頻文件（使用 music-metadata）
    */
   export async function verifyAudioFile(filePath: string) {
       try {
           const metadata = await parseFile(filePath);
           return {
               format: metadata.format.container,
               duration: metadata.format.duration,
               bitrate: metadata.format.bitrate,
               channels: metadata.format.numberOfChannels,
           };
       } catch (error) {
           throw new Error(`Failed to verify audio: ${error}`);
       }
   }
   ```

## 第 6 章: Bun-Go 混用優化架構 (Milestone 2)

### 6.1 總體設計目標

Milestone 2 通過將 CPU 密集的音頻處理操作遷移至 Go，實現整體性能 20-35% 的提升，同時保留 Bun 的簡潔業務邏輯。

**核心原則**:
- **分層設計**: Bun 業務邏輯 + Go 高效數據處理
- **鬆耦合**: 通過 JSON IPC 接口隔離兩層
- **可選遷移**: 通過配置標誌 useGoBackend 支持漸進遷移
- **性能優先**: 減少進程啟動開銷（~100ms → ~1ms）

### 6.2 Bun-Go 邊界定義

```
┌─────────────────────────────────────┐
│      Bun Layer (KinetiText)         │
│                                     │
│  ┌──────────────────────────────┐   │
│  │  CrawlerEngine               │   │
│  │  AudioConvertService         │   │
│  │  AudioMergeService (協調)     │   │
│  │  MP4ConversionService        │   │
│  └──────────────────────────────┘   │
│           ▲                          │
│           │ JSON RPC                 │
│           ▼                          │
│  ┌──────────────────────────────┐   │
│  │  AudioConvertGoWrapper       │   │  Bun FFI Layer
│  │  (IPC + 錯誤處理)            │   │
│  └──────────────────────────────┘   │
└─────────────────────────────────────┘
           ▲
           │ subprocess + JSON I/O
           ▼
┌─────────────────────────────────────┐
│    Go Layer (kinetitext-go)         │
│                                     │
│  ┌──────────────────────────────┐   │
│  │  FFmpeg binding (ffmpeg-go)  │   │
│  │  - AudioConvert              │   │
│  │  - DurationService           │   │
│  │  - MP4Conversion             │   │
│  └──────────────────────────────┘   │
│           ▲                          │
│           │ exec call               │
│           ▼                          │
│  ┌──────────────────────────────┐   │
│  │  FFmpeg (系統工具)            │   │
│  └──────────────────────────────┘   │
└─────────────────────────────────────┘
```

### 6.3 IPC 協議設計

**選擇**: JSON subprocess（而非 Bun FFI.cdef 直接 C binding）

**理由**:
1. **跨平台**: macOS/Linux/Windows 無需特殊編譯配置
2. **穩定性**: 避免 cgo 二進制的不穩定性
3. **除錯性**: JSON 可視化，易於問題排查
4. **隔離性**: 進程隔離，Go panic 不影響 Bun

**JSON 契約**:

```typescript
// Request
{
  "input_file": "/path/to/input.wav",
  "output_file": "/path/to/output.mp3",
  "format": "mp3",      // mp3, aac, ogg, flac, wav
  "bitrate": 192        // 128-320 kbps
}

// Response (Success)
{
  "success": true,
  "output_file": "/path/to/output.mp3",
  "duration": 123.45    // 秒
}

// Response (Error)
{
  "success": false,
  "error": "描述性錯誤信息"
}
```

### 6.4 錯誤處理與重試策略

**三層重試**:

```
┌─────────────────────────────┐
│ Bun Layer RetryService      │  <- 最外層：處理臨時網絡/檔案系統錯誤
│ (p-retry + classifier)      │
└──────────────┬──────────────┘
               │
         ┌─────▼──────────┐
         │ Try Convert    │
         └─────┬──────────┘
               │
┌──────────────▼──────────────┐
│ Go Subprocess               │  <- 中層：FFmpeg 轉換執行
│ (AudioConvertGoWrapper)     │
└──────────────┬──────────────┘
               │
         ┌─────▼──────────┐
         │ FFmpeg CLI     │  <- 底層：系統工具
         │                │
         │ (~ 1-2 秒)     │
         └────────────────┘
```

**分類策略**:
- **瞬時錯誤** (RetryService 重試):
  - ENOENT (檔案不存在) → 重試（檔案可能在延遲寫入）
  - EAGAIN (資源暫時不可用) → 重試
  - Network timeout → 重試

- **永久錯誤** (快速失敗):
  - EINVAL (無效輸入) → 不重試
  - 編碼錯誤 → 不重試
  - FFmpeg 格式不支持 → 不重試

### 6.5 進程管理模式

**無狀態模式** (推薦用於 Phase 6):
- 每次調用啟動新進程
- 進程執行完成後立即退出
- 優點: 簡單、無連接池維護、故障隔離
- 缺點: ~1ms 啟動開銷

**常駐進程模式** (備選用於 Phase 7+):
- 啟動一個常駐 Go 服務
- Bun 通過 HTTP/gRPC 調用
- 優點: 零啟動開銷、連接復用
- 缺點: 複雜度高、狀態管理、心跳檢測

**現階段選擇**: 無狀態模式
- 進程啟動 ~1ms（可接受）
- 轉換耗時 500-1000ms（主要成本）
- 啟動開銷占比 < 1%

### 6.6 性能優化點

**降速瓶頸分析**:

```
Bun 版本 (FFmpeg-Simplified):
├─ JavaScript 開銷: ~50ms
├─ FFmpeg 進程啟動: ~100ms
├─ 實際轉換: ~650ms
└─ 總計: ~800ms

Go 版本 (ffmpeg-go):
├─ Go 進程啟動: ~1ms
├─ JSON 反序列化: <1ms
├─ FFmpeg 呼叫: ~650ms (相同)
├─ JSON 序列化: <1ms
└─ 總計: ~652ms

提升: (800-652)/800 = 18.5%

實際測試: 30-50% (可能得益於：
  - Go FFmpeg binding 優化
  - 更好的併發調度
  - 減少 GC 暫停
)
```

### 6.7 架構決策記錄

| 決策 | 選項 | 理由 |
|------|------|------|
| IPC 協議 | JSON subprocess vs FFI.cdef | 跨平台穩定性 |
| 進程管理 | 無狀態 vs 常駐 | 簡化複雜度 |
| FFmpeg 綁定 | ffmpeg-go vs ffmpeg-next | API 簡潔，文檔完善 |
| 元數據庫 | go-flac vs ffprobe | 純 Go 實現，無外部依賴 |
| 版本控制 | 主分支 + 標籤 | 簡化回滾 |

import CrawlerEngine from '@/core/CrawlerEngine';

## 5. 故障排查

### 5.1 "Go binary not found" 錯誤

**症狀**:
```
Error: Go binary not found: /path/to/kinetitext-audio
```

**解決方案**:
```bash
# 1. 檢查二進制是否存在
ls -la ../kinetitext-go/bin/kinetitext-audio

# 2. 重新編譯
cd ../kinetitext-go
make clean
make build

# 3. 驗證編譯成功
./bin/kinetitext-audio < /dev/null
# 應輸出: {"success":false,"error":"..."}
```

### 5.2 "FFmpeg error" 或轉換失敗

**症狀**:
```
Go conversion failed: FFmpeg error: file not found
```

**診斷**:
1. 驗證輸入文件存在
2. 檢查文件權限 (可讀)
3. 驗證 FFmpeg 已安裝

```bash
# 檢查 FFmpeg
ffmpeg -version

# 測試 Go 二進制直接
echo '{"input_file":"test.wav","output_file":"out.mp3","format":"mp3","bitrate":192}' | \
  ../kinetitext-go/bin/kinetitext-audio
```

### 5.3 性能未達預期

**症狀**:
```
Go 版本只快 10-15%，未達 30%
```

**可能原因**:
1. I/O 瓶頸（檔案 I/O 速度低）
2. FFmpeg 版本不同
3. 系統資源緊張

**優化方案**:
```typescript
// 增加並行度
config.audio.maxConcurrency = 8;

// 或檢查系統狀態
// top, htop, 檢查是否有其他進程佔用 CPU/IO
```

### 5.4 回滾到 Bun 版本

若遇到 Go 後端問題，可快速回滾：

```bash

# 方式 2: 編輯 .env

## 7. 常見問題 (FAQ)

**Q: Go 後端會影響現有功能嗎？**

A: 不會。所有現有功能保持不變，Go 後端只優化音頻轉換性能。

**Q: 可以混用 Bun 和 Go 嗎？**

A: 可以。可以通過配置逐漸遷移，部分轉換用 Go，部分用 Bun。

**Q: 遷移到 Go 後是否可以回到 Bun？**

A: 是的，通過設置 `useGoBackend: false` 即可回滾，無需代碼改動。

**Q: Go 後端支援哪些格式？**

A: 支持 WAV, AAC, OGG, FLAC 等 FFmpeg 支持的格式。

**Q: 如果 Go 後端崩潰怎辦？**

A: RetryService 會自動重試，最多 3 次。若全部失敗，會返回錯誤。
