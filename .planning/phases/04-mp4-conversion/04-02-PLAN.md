```typescript
import { AudioMergeService } from './AudioMergeService'
import { MP4ConversionService } from './MP4ConversionService'
import { DurationService } from './DurationService'
import { MP4Metadata, MP4ConversionResult, GroupingReport } from '../types/audio'
import { MP4ConversionConfig, loadMP4Config } from '../config/MP4ConversionConfig'
import { getLogger } from '../utils/logger'

export interface MP4PipelineOptions {
  readonly mergedAudioDir: string // Input: directory of merged MP3 files from Phase 3
  readonly outputDir: string // Output: directory for M4A files
  readonly metadataSource?: Readonly<Record<string, MP4Metadata>> // Metadata by filename (optional)
  readonly groupingReport?: GroupingReport // Optional: use mergeBatch report for structured output
  readonly dryRun?: boolean // Preview conversion without execution
}

export interface MP4PipelineReport {
  readonly timestamp: number
  readonly inputDirectory: string
  readonly outputDirectory: string
  readonly totalFiles: number
  readonly successCount: number
  readonly failureCount: number
  readonly results: ReadonlyArray<MP4ConversionResult>
  readonly dryRun: boolean
  readonly errors: ReadonlyArray<string>
}

export class MP4Pipeline {
  constructor(
    private readonly audioMergeService: AudioMergeService,
    private readonly mp4ConversionService: MP4ConversionService,
    private readonly durationService: DurationService,
    private readonly config: MP4ConversionConfig
  ) {
    this.logger = getLogger('MP4Pipeline')
  }

  private readonly logger: any

  async execute(options: MP4PipelineOptions): Promise<MP4PipelineReport> {
    const startTime = Date.now()
    const errors: string[] = []

    try {
      // Step 1: Discover merged MP3 files
      this.logger.info({ dir: options.mergedAudioDir }, 'Scanning for merged MP3 files')
      const mergedFiles = await this.discoverMergedFiles(options.mergedAudioDir)

      if (mergedFiles.length === 0) {
        throw new Error(`No merged MP3 files found in ${options.mergedAudioDir}`)
      }

      this.logger.info({ count: mergedFiles.length }, 'Found merged files')

      // Step 2: Build conversion tasks
      const conversions = mergedFiles.map(file => ({
        inputPath: file,
        outputPath: `${options.outputDir}/${this.getOutputFilename(file)}`,
        metadata: options.metadataSource?.[this.getBasename(file)] ?? {}
      }))

      // Step 3: Dry-run preview
      if (options.dryRun) {
        this.logger.info({ count: conversions.length }, 'DRY RUN: Would convert the following files')
        conversions.forEach((conv, idx) => {
          this.logger.info(
            { index: idx, input: conv.inputPath, output: conv.outputPath },
            `  [${idx + 1}] ${this.getBasename(conv.inputPath)} → ${this.getBasename(conv.outputPath)}`
          )
        })

        return {
          timestamp: startTime,
          inputDirectory: options.mergedAudioDir,
          outputDirectory: options.outputDir,
          totalFiles: conversions.length,
          successCount: 0,
          failureCount: 0,
          results: [],
          dryRun: true,
          errors: []
        }
      }

      // Step 4: Execute conversions
      this.logger.info({ count: conversions.length }, 'Starting MP4 conversion batch')
      const results = await this.mp4ConversionService.convertBatch(conversions)

      // Step 5: Validate results
      const successCount = results.filter(r => !r.error).length
      const failureCount = results.filter(r => r.error).length

      if (failureCount > 0) {
        results
          .filter(r => r.error)
          .forEach(r => {
            errors.push(`${this.getBasename(r.inputPath)}: ${r.error}`)
          })
      }

      this.logger.info(
        { success: successCount, failure: failureCount, duration: Date.now() - startTime },
        'MP4 conversion batch complete'
      )

      return {
        timestamp: startTime,
        inputDirectory: options.mergedAudioDir,
        outputDirectory: options.outputDir,
        totalFiles: conversions.length,
        successCount,
        failureCount,
        results,
        dryRun: false,
        errors
      }
    } catch (error) {
      const msg = error instanceof Error ? error.message : 'Unknown error'
      this.logger.error({ error }, 'Pipeline execution failed')
      errors.push(msg)

      return {
        timestamp: startTime,
        inputDirectory: options.mergedAudioDir,
        outputDirectory: options.outputDir,
        totalFiles: 0,
        successCount: 0,
        failureCount: 1,
        results: [],
        dryRun: options.dryRun ?? false,
        errors
      }
    }
  }

  private async discoverMergedFiles(dir: string): Promise<string[]> {
    const dirHandle = Bun.file(dir)
    if (!(await dirHandle.exists())) {
      throw new Error(`Directory not found: ${dir}`)
    }

    const files: string[] = []
    for await (const file of Bun.file(dir).stream()) {
      // Note: Bun.file() returns a File object; use readdir-like approach
      // For now, use platform command or filesystem iteration
    }

    // Fallback: use glob pattern
    // Phase 04-02 can use fast-glob or similar; for now, iterate with Bun API
    // Placeholder: return array of .mp3 files in directory
    const files_list = (await Bun.glob('*.mp3').glob(dir)) ?? []
    return files_list
      .map(f => `${dir}/${f}`)
      .sort()
  }

  private getOutputFilename(inputPath: string): string {
    const basename = this.getBasename(inputPath)
    // Convert .mp3 to .m4a
    return basename.replace(/\.mp3$/i, '.m4a')
  }

  private getBasename(path: string): string {
    return path.split('/').pop() ?? path
  }
}
```

Key design decisions:
- Discovers merged MP3 files from directory (input from Phase 3 mergeBatch)
- Maps metadata via optional metadataSource map (keyed by filename)
- Supports --dry-run to preview without FFmpeg execution
- Uses p-limit internally via MP4ConversionService.convertBatch()
- Logs every step via Pino for observability
- Returns MP4PipelineReport with per-file results and error summary
- All data structures immutable (readonly)

Per ROADMAP: Phase 4 requirement "端到端管道整合 (Phase 1-4 完整工作流)"

```typescript
#!/usr/bin/env bun
import { MP4Pipeline } from '../src/core/services/MP4Pipeline'
import { MP4ConversionService } from '../src/core/services/MP4ConversionService'
import { AudioMergeService } from '../src/core/services/AudioMergeService'
import { DurationService } from '../src/core/services/DurationService'
import { RetryService } from '../src/core/services/RetryService'
import { AudioErrorClassifier } from '../src/core/services/AudioErrorClassifier'
import { loadMP4Config } from '../src/core/config/MP4ConversionConfig'
import { getLogger } from '../src/core/utils/logger'
import { MP4Metadata } from '../src/core/types/audio'

const logger = getLogger('mp3_to_mp4')

interface CliArgs {
  input: string // --input=/path/to/merged/mp3s
  output: string // --output=/path/to/m4a/output
  metadata?: string // --metadata=/path/to/metadata.json (optional)
  dryRun?: boolean // --dry-run (optional)
}

function parseArgs(): CliArgs {
  const args = process.argv.slice(2)
  const parsed: Record<string, string | boolean> = {}

  args.forEach(arg => {
    if (arg.startsWith('--')) {
      const [key, value] = arg.substring(2).split('=')
      parsed[key] = value ?? true
    }
  })

  if (!parsed.input || !parsed.output) {
    console.error('Usage: bun scripts/mp3_to_mp4.ts --input=/path --output=/path [--metadata=/path] [--dry-run]')
    process.exit(1)
  }

  return {
    input: parsed.input as string,
    output: parsed.output as string,
    metadata: parsed.metadata as string | undefined,
    dryRun: parsed['dry-run'] === true || parsed['dry-run'] === 'true'
  }
}

function formatReport(report: any): string {
  const lines: string[] = [
    '',
    '═══════════════════════════════════════',
    'MP3 → M4A 轉換報告',
    '═══════════════════════════════════════',
    `時間戳: ${new Date(report.timestamp).toISOString()}`,
    `輸入目錄: ${report.inputDirectory}`,
    `輸出目錄: ${report.outputDirectory}`,
    `總檔案數: ${report.totalFiles}`,
    `成功: ${report.successCount}`,
    `失敗: ${report.failureCount}`,
    `乾運行模式: ${report.dryRun ? '是' : '否'}`,
    ''
  ]

  if (report.dryRun) {
    lines.push('預覽模式 — 未執行實際轉換')
    lines.push('若要開始轉換，請執行:')
    lines.push(`  bun scripts/mp3_to_mp4.ts --input=${report.inputDirectory} --output=${report.outputDirectory}`)
  }

  if (report.errors.length > 0) {
    lines.push('錯誤:')
    report.errors.forEach(err => {
      lines.push(`  • ${err}`)
    })
  }

  if (report.results.length > 0 && !report.dryRun) {
    lines.push('轉換結果:')
    report.results.forEach((result, idx) => {
      const status = result.error ? '❌' : '✅'
      const basename = result.inputPath.split('/').pop()
      const duration = result.duration ? `(${Math.round(result.duration)}s)` : ''
      const errMsg = result.error ? ` — ${result.error}` : ''
      lines.push(`  ${status} [${idx + 1}] ${basename} ${duration}${errMsg}`)
    })
  }

  lines.push('═══════════════════════════════════════')
  lines.push('')

  return lines.join('\n')
}

async function main(): Promise<void> {
  try {
    const args = parseArgs()
    logger.info({ args }, 'Starting MP3→M4A conversion')

    // Load configuration
    const config = await loadMP4Config()

    // Load metadata if provided
    let metadataMap: Record<string, MP4Metadata> | undefined
    if (args.metadata) {
      const metadataFile = Bun.file(args.metadata)
      if (await metadataFile.exists()) {
        const json = await metadataFile.json()
        metadataMap = json
        logger.info({ count: Object.keys(metadataMap).length }, 'Loaded metadata map')
      } else {
        logger.warn({ file: args.metadata }, 'Metadata file not found, proceeding without')
      }
    }

    // Initialize services
    const retryService = new RetryService()
    const errorClassifier = new AudioErrorClassifier()
    const durationService = new DurationService()
    const audioMergeService = new AudioMergeService()
    const mp4ConversionService = new MP4ConversionService(
      config,
      retryService,
      errorClassifier
    )

    // Create and execute pipeline
    const pipeline = new MP4Pipeline(
      audioMergeService,
      mp4ConversionService,
      durationService,
      config
    )

    const report = await pipeline.execute({
      mergedAudioDir: args.input,
      outputDir: args.output,
      metadataSource: metadataMap,
      dryRun: args.dryRun
    })

    // Print report
    console.log(formatReport(report))

    // Exit with appropriate code
    process.exit(report.failureCount > 0 ? 1 : 0)
  } catch (error) {
    logger.error({ error }, 'Fatal error')
    console.error(`Error: ${error instanceof Error ? error.message : 'Unknown error'}`)
    process.exit(1)
  }
}
```
