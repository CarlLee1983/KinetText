    ```typescript
    import { $ } from 'bun'
    import { parseFile } from 'music-metadata'
    import path from 'node:path'
    import { unlink } from 'node:fs/promises'
    import pLimit from 'p-limit'
    import { RetryService } from './RetryService'
    import { AudioErrorClassifier } from './AudioErrorClassifier'
    import { AudioConvertConfig } from '../../config/AudioConvertConfig'
    import { RetryConfig } from '../../config/RetryConfig'
    import { createLogger } from '../utils/logger'
    import type {
      ConversionResult,
      ConversionBatchResult,
      ConversionError,
      AudioMetadata,
    } from '../types/audio'

    export class AudioConvertService {
      private readonly config: AudioConvertConfig
      private readonly retryService: RetryService
      private readonly logger: pino.Logger

      constructor(
        config: AudioConvertConfig = new AudioConvertConfig(),
        retryService?: RetryService
      ) {
        this.config = config
        const retryConfig = new RetryConfig({
          timeoutMs: config.ffmpegTimeoutMs,
          operationTimeoutMs: config.ffmpegTimeoutMs * 2,
        })
        // Use AudioErrorClassifier so FFmpeg errors are properly classified
        this.retryService = retryService ?? new RetryService(retryConfig)
        this.logger = createLogger('audio-convert')
      }

      static async checkFfmpegAvailable(): Promise<void> { ... }

      async convertToMp3(inputPath: string, outputPath: string): Promise<ConversionResult> { ... }

      async convertBatch(files: ReadonlyArray<{ input: string; output: string }>): Promise<ConversionBatchResult> { ... }

      async getMetadata(filePath: string): Promise<AudioMetadata> { ... }

      private async runFfmpegConversion(inputPath: string, outputPath: string): Promise<void> { ... }

      private async cleanupPartialFile(filePath: string): Promise<void> { ... }

      private detectFormat(filePath: string): string { ... }
    }
    ```

    Key implementation details:
    - `checkFfmpegAvailable()`: runs `ffmpeg -version` via `Bun.$`, throws Error('FFmpeg not found in PATH. Install: brew install ffmpeg') on failure
    - `convertToMp3()`: wraps `runFfmpegConversion` in `this.retryService.execute()`, measures durationMs, reads output metadata via `getMetadata()`, returns immutable ConversionResult. On failure, calls `cleanupPartialFile()`.
    - `runFfmpegConversion()`: executes `ffmpeg -y -i ${inputPath} -codec:a libmp3lame -b:a ${this.config.bitrate} -ar ${this.config.sampleRate} ${outputPath}` via `Bun.$`. Checks exitCode !== 0 and throws Error with stderr excerpt (first 200 chars).
    - `convertBatch()`: uses `pLimit(this.config.maxConcurrency)`, maps files through `convertToMp3()` via `Promise.allSettled()`, aggregates into ConversionBatchResult. Each failure becomes a ConversionError, does NOT abort other conversions.
    - `getMetadata()`: calls `parseFile(filePath)` from music-metadata, returns immutable AudioMetadata object.
    - `detectFormat()`: extracts extension from path, maps to format name (e.g., '.wav' -> 'WAV', '.aac' -> 'AAC').
    - `cleanupPartialFile()`: attempts `unlink(filePath)`, silently catches ENOENT.
